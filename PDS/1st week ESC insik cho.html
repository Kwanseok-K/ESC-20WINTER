    output = np.empty(len(values))
    for i in range(len(values)):

sorting arrays¶

In [184]:
def selection_sort(x):
    for i in range(len(x)):
        swap = i + np.argmin(x[i:])
        (x[i], x[swap]) = (x[swap], x[i])
    return x
